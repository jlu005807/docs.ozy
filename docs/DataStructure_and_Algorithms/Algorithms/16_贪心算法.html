  • ж №жҚ®йўҳж„ҸйҖүеҸ–дёҖз§ҚйҮҸеәҰж ҮеҮҶпјӣ

жҙ»еҠЁйҖүжӢ©й—®йўҳ

жҙ»еҠЁйҖүжӢ©й—®йўҳ(activity-selection problem)пјҡжҳҜдёҖдёӘи°ғеәҰеӨҡдёӘжҙ»еҠЁй—®йўҳпјҢиҝҷдәӣжҙ»еҠЁйңҖиҰҒдә’ж–Ҙз«һдәүе…ұдә«иө„жәҗгҖӮд»ҺдёҖдёӘжҙ»еҠЁж—ҘзЁӢиЎЁдёӯйҖүеҮәи§„жЁЎжңҖеӨ§зҡ„зӣёдә’е…је®№зҡ„жҙ»еҠЁ(activity)зҡ„йӣҶеҗҲгҖӮ

int GREEDY-KNAPSACK(vector<int> weight,vector<int>value,int bagweight)
{
    int max_value=0;
    int now_weight=bagweight;
    vector<int> object(weight.size());//и®°еҪ•зү©е“Ғж”ҫе…ҘжҜ”дҫӢ
    for(int i=0;i<weight.size();i++){
        //ж— жі•иЈ…е…Ҙе®Ңж•ҙзү©е“Ғ
        if(weight[i]>now_weight)
        {
            //жҢүжҜ”дҫӢж”ҫе…Ҙ
            object[i]=now_weight/weight[i];
            max_value+=value[i]*object[i];
            break;
        }
        max_value+=value[i];
        now_weight-=weight[i];
        object[i]=1;
    }
    return max_value;
}

void JS(std::vector<int>& D, std::vector<int>& J, int n, int& k) {
    D[0] = 0;
    J[0] = 0;
    k = 1;
    J[1] = 1;
    //еӨ„зҗҶз¬¬iдёӘдҪңдёҡ
    for (int i = 2; i <= n; ++i) {
        int r = k;
        //жүҫеҲ°жҸ’е…Ҙзҡ„дҪҚзҪ®r
        while (D[J[r]] > D[i] && D[J[r]] != r) {
            --r;
        }
        //иЎЁзӨәжүҫеҲ°дәҶжҸ’е…ҘдҪҚзҪ®r
        if (D[J[r]] <= D[i] && D[i] > r) {
            //е®һзҺ°дҪңдёҡr+1еҲ°дҪңдёҡkдҫқж¬ЎеҫҖеҗҺз§»еҠЁдёҖдёӘдҪҚзҪ®
            for (int l = k; l >= r + 1; --l) {
                J[l + 1] = J[l];
            }
            //жҸ’е…Ҙ
            J[r + 1] = i;
            ++k;
        }
    }
}

int find(std::vector<int>& parent, int x) {
    return parent[x]>0?x:parent[x]=find(parent,parent[x]);
}

void UNION(std::vector<int>& parent,int i,int j)
{
    int x=parent[i]+parent[j];
    if(parent[i]>parent[j]){
        parent[i]=j;
        parent[j]=x;
    }else {
        parent[j]=i;
        parent[i]=x;
    }
}

void FJS(std::vector<int>& D, int n, int b, std::vector<int>& J, int& k) {
    int b = min(n,*max_element(D.begin(),D.end()));
    // еҲқе§ӢеҢ–Fж•°з»„пјҢF[0..b]д»ЈиЎЁж—¶й—ҙж§Ҫзҡ„зҲ¶жҢҮй’Ҳ
    std::vector<int> F(b + 1);
    for (int i = 0; i <= b; ++i) {
        F[i] = i;
    }

    k = 0; // еҸҜиЎҢдҪңдёҡж•°
    J.resize(n + 1); // J[1..k]еӯҳеӮЁеҸҜиЎҢдҪңдёҡ

    // жҢүеҲ©ж¶ҰйҷҚеәҸеӨ„зҗҶдҪңдёҡпјҲеҒҮи®ҫиҫ“е…Ҙе·ІжҺ’еәҸпјү
    for (int i = 1; i <= n; ++i) {
        int d = std::min(n, D[i]); // и®Ўз®—жңүж•ҲжҲӘжӯўж—¶й—ҙ
        int j = find(F, d); // жүҫеҲ°жңҖжҷҡеҸҜз”Ёж—¶й—ҙж§Ҫ

        if (F[j] != 0) { // еӯҳеңЁеҸҜз”Ёж—¶й—ҙж§Ҫ
            J[++k] = i; // е°ҶдҪңдёҡiеҠ е…Ҙи§ЈйӣҶ
            int l = find(F, F[j] - 1); // жүҫеҲ°дёӢдёҖдёӘеҸҜз”ЁдҪҚзҪ®
            UNION(F,l,j);//еҗҲе№¶йӣҶеҗҲ
            F[j] = F[l]; // еҗҲе№¶ж—¶й—ҙж§ҪпјҢи·Ҝеҫ„еҺӢзј©
        }
    }
}

    //е…ӢйІҒж–ҜеҚЎе°” (Kruskal)з®—жі•пјҢеҸҜз§°дёәвҖңеҠ иҫ№жі•вҖқпјҢйҖӮз”ЁдәҺзЁҖз–Ҹеӣҫ
    //жҜҸж¬ЎйҖүеҮәжқғеҖјжңҖе°Ҹе№¶дё”ж— жі•дҪҝзҺ°жңүзҡ„ж ‘еҪўжҲҗзҺҜзҡ„иҫ№еҠ е…ҘжңҖе°Ҹж”Ҝж’‘ж ‘,иҝ”еӣһдёҖдёӘеӣҫ
    std::vector<Edge> MiniSpanTree_Kruskal(std::vector<Edge> graph,int number)
    {
        //йқһиҝһйҖҡеӣҫ
        if (Connected_Component(graph,number) != 1)
        {
            return{};
        }
        std::vector<Edge> MiniSpanTree;
        //еңЁEdgeеўһеҠ дәҶweightжҲҗе‘ҳеӯҳеӮЁжқғеҖјпјҢеҸҜд»ҘзӣҙжҺҘз”ЁsortжҺ’еәҸ
        std::sort(graph.begin(), graph.end(), [](const Edge& a, const Edge& b)->bool {return a.weight < b.weight; });

        //иҫ…еҠ©ж•°з»„Vexset,ж ҮиҜҶеҗ„дёӘйЎ¶зӮ№жүҖеұһзҡ„иҝһйҖҡеҲҶйҮҸ,зұ»дјјдәҺе№¶жҹҘйӣҶ
        std::vector<int> Vexset;
        Vexset.resize(number,0);
        //еҲқе§ӢеҢ–,иЎЁзӨәеҗ„йЎ¶зӮ№иҮӘжҲҗдёҖдёӘиҝһйҖҡеҲҶеқҖ
        for (int i = 0; i < number; i++)
        {
            Vexset[i] = i;
        }

        //ејҖе§ӢеҲӣе»әжңҖе°Ҹж”Ҝж’‘ж ‘
        for (int i = 0; i < graph.size(); i++)
        {
            int v1 = graph[i].u;
            int v2 = graph[i].v;
            int vs1 = Vexset[v1];
            int vs2 = Vexset[v2];

            //иҫ№зҡ„дёӨдёӘйЎ¶зӮ№дёҚеңЁеҗҢдёҖиҝһйҖҡеҲҶйҮҸ
            if (vs1 != vs2)
            {
                //еҠ е…ҘжӯӨиҫ№
                MiniSpanTree.push_back(Edge(v1, v2));

                //еҗҲе№¶vs1е’Ңvs2дёӨдёӘеҲҶйҮҸпјҢеҚідёӨдёӘйӣҶеҗҲз»ҹдёҖзј–еҸ·
                for (int j = 0; j < number; j++)
                {
                    if (Vexset[j] == vs2)Vexset[j] = vs1;
                }
            }
        }

        //иҝ”еӣһ
        return MiniSpanTree;
    }

// жҷ®йҮҢе§Ҷз®—жі•дјҳеҢ–пјҢдҪҝз”Ёдјҳе…ҲйҳҹеҲ—
Adj_Matrix MiniSpanTree_Prim(const Adj_Matrix& adj, int u = 0 /*иө·е§ӢзӮ№*/) {
    // йқһиҝһйҖҡеӣҫпјҢиҝ”еӣһз©әеӣҫ
    if (Connected_Component(adj) != 1) {
        return Adj_Matrix(0);
    }

    int n = adj.graph.size();  // еӣҫзҡ„йЎ¶зӮ№ж•°
    Adj_Matrix MiniSpanTree(n, adj.is_direct);  // жңҖе°Ҹз”ҹжҲҗж ‘

    // и®°еҪ•йЎ¶зӮ№жҳҜеҗҰе·Із»ҸеҠ е…Ҙз”ҹжҲҗж ‘
    std::vector<bool> vis(n, false);

    // дҪҝз”Ёдјҳе…ҲйҳҹеҲ—пјҲжңҖе°Ҹе ҶпјүдҝқеӯҳжңӘеҠ е…Ҙз”ҹжҲҗж ‘зҡ„зӮ№е’Ңе…¶еҜ№еә”зҡ„жңҖе°Ҹиҫ№жқғеҖј
    // йҳҹеҲ—дёӯеӯҳеӮЁзҡ„жҳҜ (иҫ№жқғеҖј, йЎ¶зӮ№) зҡ„ pair
    std::priority_queue<std::pair<int, int>, std::vector<std::pair<int, int>>, std::greater<>> pq;

    // еҲқе§ӢеҢ–пјҡд»Һиө·е§ӢзӮ№uејҖе§ӢпјҢеҠ е…ҘйҳҹеҲ—пјҢжқғеҖјдёә0
    pq.push({0, u});

    // йҖүжӢ©е…¶д»–n-1дёӘйЎ¶зӮ№пјҢз”ҹжҲҗn-1жқЎиҫ№
    while (!pq.empty()) {
        // еҸ–еҮәеҪ“еүҚжқғеҖјжңҖе°Ҹзҡ„иҫ№
        int current_weight = pq.top().first;
        int u0 = pq.top().second;
        pq.pop();

        // еҰӮжһңйЎ¶зӮ№u0е·Із»ҸеҠ е…Ҙз”ҹжҲҗж ‘пјҢеҲҷи·іиҝҮ
        if (vis[u0]) continue;

        // ж Үи®°u0дёәе·Іи®ҝй—®
        vis[u0] = true;

        // жӣҙж–°жңҖе°Ҹз”ҹжҲҗж ‘зҡ„иҫ№
        if (u0 != u) {  // жҺ’йҷӨиө·е§ӢзӮ№
            int v0 = pq.top().second;  // йЎ¶зӮ№u0зҡ„еүҚй©ұ
            MiniSpanTree.graph[u0][v0] = current_weight;

            // ж— еҗ‘еӣҫеҲҷеўһеҠ еҜ№з§°иҫ№
            if (!MiniSpanTree.is_direct) {
                MiniSpanTree.graph[v0][u0] = current_weight;
            }
        }

        // жӣҙж–°дёҺu0зҡ„зӣёйӮ»иҠӮзӮ№зҡ„жңҖе°Ҹиҫ№пјҢе№¶еҠ е…Ҙдјҳе…ҲйҳҹеҲ—
        for (int i = 0; i < n; i++) {
            if (!vis[i] && adj.graph[u0][i] < INT_MAX) {  // еҰӮжһңiжңӘи®ҝй—®дё”жңүиҫ№
                pq.push({adj.graph[u0][i], i});
            }
        }
    }

    // иҝ”еӣһз”ҹжҲҗзҡ„жңҖе°Ҹз”ҹжҲҗж ‘
    return MiniSpanTree;
}

vector<int> GreedyIntervalSelector(int m, int d, vector<int>& x) {
    vector<int> S;
    int n = x.size();
    if (n == 0 || x[0] > m) return {}; // еҲқе§ӢдҪҚзҪ®дёҚеҸҜиҫҫз¬¬дёҖдёӘзӮ№

    int s = 0;       // еҪ“еүҚе·ІеҲ°иҫҫзҡ„жңҖиҝңдҪҚзҪ®
    int current = 0; // еҪ“еүҚйҖүдёӯзҡ„зӮ№зҙўеј•

    while (s + m < d) {
        int prev = s;
        int left = current + 1;
        int right = n - 1;
        int best = -1;

        // дәҢеҲҶжҹҘжүҫжңҖеӨ§еҸҜиҫҫзӮ№
        while (left <= right) {
            int mid = left + (right - left) / 2;
            if (x[mid] <= prev + m) {
                best = mid;
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }

        if (best == -1) return {}; // жІЎжңүеҸҜиҫҫзӮ№

        S.push_back(best);
        s = x[best];
        current = best;
    }
    return S;
}

  • зј–з Ғ(encoding)пјҡд»»дҪ•дәҢиҝӣеҲ¶еӯ—з¬Ұзј–з Ғзҡ„зј–з ҒеҫҲз®ҖеҚ•пјҢзӣҙжҺҘе°Ҷеӯ—з¬Ұзҡ„з Ғеӯ—жҢүз…§йЎәеәҸжӢјжҺҘгҖӮ
  • и§Јз Ғ(decoding)пјҡз”ұдәҺд»»дҪ•дёҖдёӘз Ғеӯ—йғҪдёҚжҳҜе…¶е®ғз Ғеӯ—зҡ„еүҚзјҖзҡ„зј–з ҒпјҢжүҖд»ҘжҜҸдёӘеӯ—з¬Ұзҡ„зј–з ҒйғҪжҳҜж— жӯ§д№үзҡ„гҖӮжүҖд»ҘеҸҜд»ҘжҢүз…§йЎәеәҸиҜҶеҲ«з Ғеӯ—пјҢе№¶е°Ҷе…¶иҪ¬жҚўдёәеҺҹжқҘеӯ—з¬ҰгҖӮ

        allocate a new node x
        x.freq = 0
        INSERT(PQ, x)
    for i = 1 to (n + Ој - 1) / 2
        allocate a new node w
        x = EXTRACT-MIN(PQ)
        y = EXTRACT-MIN(PQ)
        z = EXTRACT-MIN(PQ)
        w.first = x
        w.second = y
        w.third = z
        w.freq = x.freq + y.freq + z.freq
        INSERT(PQ, w)
    return EXTRACT-MIN(PQ)    // the root of the tree is the only node left

йңҖиҰҒеҗҲе№¶ (n+ОјвҲ’1)/2 ж¬ЎпјҢжҜҸж¬Ўдјҳе…ҲйҳҹеҲ—и°ғж•ҙзҡ„иҝҗиЎҢж—¶й—ҙдёә O(lgвҒЎn) пјҢжүҖд»ҘOPTIMAL-TERNARY-MERGEзҡ„иҝҗиЎҢж—¶й—ҙдёә O(nlgвҒЎn) гҖӮ

иҜ„пјҡеҸӮиҖғгҖҠж•°жҚ®з»“жһ„пјҲCиҜӯиЁҖзүҲпјүгҖӢ11.5 жңҖдҪіеҪ’е№¶ж ‘гҖӮ

зҰ»зәҝзј“еӯҳ

[!note]

зј“еӯҳ(cache)пјҡе®№йҮҸжҜ”дё»еӯҳе°ҸпјҢйҖҹеәҰжҜ”дё»еӯҳеҝ«зҡ„еӯҳеӮЁеҷЁ(memory)гҖӮи®Ўз®—жңәйҖҡиҝҮжҠҠйңҖиҰҒи®ҝй—®зҡ„ж•°жҚ®зҡ„дёҖйғЁеҲҶеӮЁеӯҳеңЁзј“еӯҳдёӯпјҢеҸҜд»ҘеҮҸе°‘ж•°жҚ®зҡ„и®ҝй—®ж—¶й—ҙгҖӮзј“еӯҳе°Ҷж•°жҚ®жңүз»„з»Үең°еӯҳж”ҫеңЁзј“еӯҳеқ—(cache blocks)дёӯпјҢзј“еӯҳеқ—еӨ§е°ҸдёҖиҲ¬жҳҜ 32 гҖҒ 64 жҲ– 128 еӯ—иҠӮгҖӮ

дё»еӯҳ(main memory)пјҡиҷҡжӢҹеҶ…еӯҳзі»з»ҹдёӯпјҢдё»еҶ…еӯҳеҸҜд»Ҙиў«и§Ҷдёәй©»з•ҷеңЁзЈҒзӣҳдёҠзҡ„ж•°жҚ®зҡ„зј“еӯҳгҖӮиҝҷдәӣдё»еӯҳеқ—(blocks)иў«з§°дёәйЎө(pages)пјҢйЎөеӨ§е°ҸдёҖиҲ¬жҳҜ 4096 еӯ—иҠӮгҖӮ

й—®йўҳ

еҪ“дёҖдёӘзЁӢеәҸжү§иЎҢж—¶пјҢйңҖиҰҒиҝӣиЎҢдёҖзі»еҲ—зҡ„еӯҳеӮЁеҷЁиҜ·жұӮгҖӮеҒҮи®ҫжңү n дёӘи®ҝеӯҳиҜ·жұӮпјҢиҝҷдәӣж•°жҚ®жҢүз…§иҜ·жұӮйЎәеәҸеҲҶеҲ«еңЁ b1,b2,вҖҰ,bn еқ—дёӯгҖӮдәӢе®һдёҠпјҢиҝҷдәӣиҜ·жұӮдёҚдјҡе®Ңе…ЁдёҚеҗҢпјҢеӨҡдёӘиҜ·жұӮжңүеҸҜиғҪйңҖиҰҒи®ҝй—®еҗҢдёҖдёӘеқ—гҖӮ

еҪ“йңҖиҰҒи®ҝй—® bi ж—¶пјҢдјҡеҮәзҺ°д»ҘдёӢдёүз§Қжғ…еҶөпјҡ

  • жғ…еҶөдёҖпјҡз”ұдәҺд№ӢеүҚи®ҝй—®иҝҮ bi пјҢ bi е·Із»ҸеңЁзј“еӯҳдёӯпјҢеҪ“йңҖиҰҒеҶҚж¬Ўи®ҝй—® bi ж—¶пјҢеҸҜд»ҘзӣҙжҺҘи®ҝй—®зј“еӯҳпјҢз§°дёәзј“еӯҳе‘Ҫдёӯ(cache hit)гҖӮ
  • жғ…еҶөдәҢпјҡbi дёҚеңЁзј“еӯҳдёӯпјҢзј“еӯҳжңӘж»ЎпјҢеҪ“йңҖиҰҒи®ҝй—® bi ж—¶пјҢзӣҙжҺҘе°Ҷ bi еЎ«е……еҲ°з©әй—Ізҡ„зј“еӯҳеқ—гҖӮ
  • жғ…еҶөдёүпјҡbi дёҚеңЁзј“еӯҳдёӯпјҢзј“еӯҳе·Іж»ЎпјҢеҪ“йңҖиҰҒи®ҝй—® bi ж—¶пјҢйңҖиҰҒйў„е…Ҳе°ҶжҹҗдёҖдёӘзј“еӯҳеқ—з©әеҮәжқҘпјҢз„¶еҗҺе°Ҷ bi еЎ«е……еҲ°з©әй—Ізҡ„зј“еӯҳеқ—гҖӮ

жғ…еҶөдәҢе’Ңжғ…еҶөдёүз§°дёәзј“еӯҳжңӘе‘Ҫдёӯ(cache miss)гҖӮжғ…еҶөдәҢз§°дёәејәеҲ¶жңӘе‘Ҫдёӯ(compulsory miss)гҖӮ

дёҖиҲ¬жғ…еҶөдёӢпјҢз”ұдәҺи®Ўз®—жңәж— жі•зҹҘйҒ“жңӘжқҘзҡ„иҜ·жұӮпјҢеӣ жӯӨзј“еӯҳ(caching)жҳҜдёҖдёӘеңЁзәҝй—®йўҳгҖӮиҝҷйҮҢжҲ‘д»¬д»…д»…иҖғиҷ‘зј“еӯҳй—®йўҳзҡ„зҰ»зәҝзүҲжң¬пјҢеҚіе·ІзҹҘе®Ңж•ҙзҡ„и®ҝеӯҳиҜ·жұӮеәҸеҲ—е’Ңзј“еӯҳеқ—ж•°йҮҸпјҢжҲ‘д»¬зҡ„зӣ®ж ҮжҳҜжңҖе°ҸеҢ–зј“еӯҳжңӘе‘ҪдёӯпјҢжңҖеӨ§еҢ–зј“еӯҳе‘ҪдёӯгҖӮ

[!important]

жҲ‘д»¬йҮҮз”Ёз§°дёәе°ҶжқҘжңҖд№…(furthest-in-future)зҡ„иҙӘеҝғзӯ–з•ҘжұӮи§ЈзҰ»зәҝзј“еӯҳй—®йўҳпјҢеҚіеҰӮжһңзј“еӯҳе·Іж»ЎпјҢйӮЈд№ҲйҖүжӢ©е·ІеңЁзј“еӯҳдёӯдё”и®ҝеӯҳеәҸеҲ—дёӯе°ҶжқҘжңҖд№…еҲ°иҫҫзҡ„еқ—иҝӣиЎҢзҪ®жҚўгҖӮеңЁжңүдәӣжғ…еҶөдёӢпјҢжҲ‘д»¬еҸҜд»ҘжҸҗеүҚзҹҘйҒ“е®Ңж•ҙзҡ„и®ҝеӯҳиҜ·жұӮеәҸеҲ—пјҢдҫӢеҰӮпјҢжҲ‘д»¬еҸҜд»Ҙе°Ҷдё»еӯҳи§ҶдёәзЈҒзӣҳжҲ–иҖ…еӣәжҖҒзЎ¬зӣҳдёӯжүҖжңүж•°жҚ®зҡ„зј“еӯҳпјҢеҚіе·Із»ҸжңүдёҖдәӣз®—жі•еҸҜд»Ҙйў„е…Ҳи®ЎеҲ’ж•ҙдёӘиҜ»еҶҷж“ҚдҪңйӣҶгҖӮжӯӨеӨ–пјҢжҲ‘д»¬еҸҜд»ҘдҪҝз”ЁжңҖдјҳз®—жі•зҡ„зј“еӯҳжңӘе‘Ҫдёӯж•°дҪңдёәжҜ”иҫғеӨҡз§ҚеңЁзәҝз®—жі•жҖ§иғҪзҡ„еҹәзәҝгҖӮиҝҷдәӣжҲ‘д»¬е°ҶеңЁ27.3иҠӮдёӯиҝӣиЎҢиҜҙжҳҺгҖӮ

зҰ»зәҝзј“еӯҳд№ҹеҸҜд»Ҙз”ЁдәҺзҺ°е®һз”ҹжҙ»дёӯзҡ„й—®йўҳзҡ„е»әжЁЎгҖӮ

и§Јзӯ”

йҮҮз”Ёз§°дёәе°ҶжқҘжңҖд№…(furthest-in-future)зҡ„иҙӘеҝғзӯ–з•ҘжұӮи§ЈзҰ»зәҝзј“еӯҳй—®йўҳпјҢеҚіеҰӮжһңзј“еӯҳе·Іж»ЎпјҢйӮЈд№ҲйҖүжӢ©е·ІеңЁзј“еӯҳдёӯдё”и®ҝеӯҳеәҸеҲ—дёӯе°ҶжқҘжңҖд№…еҲ°иҫҫзҡ„еқ—иҝӣиЎҢзҪ®жҚўгҖӮ

еҲҷжҺҘдёӢжқҘи®Ёи®әиҙӘеҝғзӯ–з•Ҙзҡ„жңҖдјҳеӯҗз»“жһ„е’ҢиҙӘеҝғйҖүжӢ©жҖ§иҙЁ

зҰ»зәҝзј“еӯҳзҡ„жңҖдјҳеӯҗз»“жһ„(Optimal substructure of offline caching)

  • е®ҡд№үеӯҗй—®йўҳ (C,i) пјҢзј“еӯҳеқ—йӣҶеҗҲдёә C пјҢи®ҝеӯҳиҜ·жұӮеәҸеҲ—дёә bi,bi+1,вҖҰ,bn пјҢеҪ“еүҚйңҖиҰҒиҜ·жұӮ bi пјҢи®ҫеҸҜз”Ёзҡ„зј“еӯҳеқ—ж•°йҮҸжңҖеӨҡдёә k пјҢ |C|вүӨk гҖӮеӯҗй—®йўҳ (C,i) зҡ„жңҖдјҳи§ЈиҰҒжұӮдҪҝеҫ—зј“еӯҳжңӘе‘Ҫдёӯж•°жңҖе°Ҹ

  • и®ҫеӯҗй—®йўҳ (C,i) зҡ„жңҖдјҳи§Јдёә S пјҢиҜ·жұӮ bi е®ҢжҲҗеҗҺзј“еӯҳеқ—йӣҶеҗҲдёә CвҖІ пјҢеӯҗй—®йўҳ (CвҖІ,i+1) зҡ„жңҖдјҳи§Јдёә SвҖІ гҖӮ

  • е®ҡд№ү RC,i дёәеӯҗй—®йўҳ (C,i) иҜ·жұӮ bi е®ҢжҲҗеҗҺеҸҜиғҪзҡ„зј“еӯҳеқ—йӣҶеҗҲзҡ„йӣҶеҗҲпјҢжңүпјҡ

    • жғ…еҶөдёҖпјҡ RC,i={C}пјҢеҜ№еә”зј“еӯҳе‘Ҫдёӯ
    • жғ…еҶөдәҢпјҡ RC,i={CвҲӘ{bi}}пјҢеҜ№еә”зј“еӯҳејәеҲ¶жңӘе‘Ҫдёӯ
    • жғ…еҶөдёүпјҡ RC,i={(CвҲ’x)вҲӘ{bi}:xвҲҲC}пјҢеҜ№еә”зј“еӯҳжңӘе‘Ҫдёӯдё”ж»Ў

    е®ҡд№ү miss(C,i) дёәеӯҗй—®йўҳ (C,i) зҡ„жңҖдјҳи§Јзҡ„зј“еӯҳжңӘе‘Ҫдёӯж•°гҖӮйҖ’еҪ’ејҸеҰӮдёӢпјҡ

зҰ»зәҝзј“еӯҳзҡ„иҙӘеҝғйҖүжӢ©жҖ§иҙЁ(Greedy-choice property of offline caching)

жңҖдјҳзҰ»зәҝзј“еӯҳе…·жңүиҙӘеҝғйҖүжӢ©жҖ§иҙЁ(Optimal offline caching has the greedy-choice property)

иҖғиҷ‘еӯҗй—®йўҳ (C,i) пјҢзј“еӯҳеқ—йӣҶеҗҲдёә C жңү k дёӘзј“еӯҳеқ—пјҢеҚіжӯӨж—¶зј“еӯҳе·Іж»ЎпјҢдё”еҮәзҺ°зј“еӯҳжңӘе‘ҪдёӯгҖӮеҪ“еүҚйңҖиҰҒиҜ·жұӮ bi пјҢи®ҫ z=bm дёә C дёӯе°ҶжқҘжңҖд№…иў«иҜ·жұӮзҡ„зј“еӯҳеқ—пјҢпјҲеҰӮжһңжңүзј“еӯҳеқ—е°ҶжқҘдёҚеҶҚиў«и®ҝй—®пјҢдјҳе…ҲиҖғиҷ‘иҝҷж ·зҡ„зј“еӯҳеқ—жҲҗдёә z пјҢжүҖд»ҘеҸҜд»ҘеўһеҠ иҷҡзј“еӯҳеқ—пјҢдҪҝеҫ— z=bm=bn+1 гҖӮпјүжҲ‘д»¬еҸҜд»Ҙз§»йҷӨзј“еӯҳеқ— z пјҢз„¶еҗҺе°ҶеўһеҠ зј“еӯҳеқ— bi пјҢеҰӮжӯӨж“ҚдҪңеҸҜд»Ҙеҫ—еҲ°еӯҗй—®йўҳ (C,i) зҡ„жҹҗдёҖдёӘжңҖдјҳи§ЈгҖӮ

иҜҒжҳҺиҜҰи§Ғз®—жі•еҜји®әпјҲз¬¬еӣӣзүҲпјүз¬¬еҚҒдә”з« пјҡиҙӘеҝғз®—жі•гҖҖз¬¬еӣӣиҠӮпјҡзҰ»зәҝзј“еӯҳ - зҹҘд№Һ

иҙӘеҝғз®—жі•еҺҹзҗҶ

[!note]

д»ҘйҖҡиҝҮиҙӘеҝғйҖүжӢ©жқҘж”№иҝӣжңҖдјҳеӯҗз»“жһ„пјҢдҪҝеҫ—йҖүжӢ©еҗҺеҸӘз•ҷдёӢдёҖ дёӘеӯҗй—®йўҳ

д№ҹе°ұжҳҜдјҳеҢ–еҠЁжҖҒи§„еҲ’зҡ„еӯҗй—®йўҳж•°йҮҸ

  • д»ҘдёӢжӯҘйӘӨи®ҫи®ЎиҙӘеҝғз®—жі•гҖӮ

[!important]

  1. жңҖдјҳеҢ–й—®йўҳдёәдҪ еҒҡеҮәйҖүжӢ©еҗҺеҸӘжңүдёҖдёӘеӯҗй—®йўҳйңҖиҰҒжұӮи§ЈгҖӮ
  2. иҜҒжҳҺжҜҸж¬ЎеҒҡеҮәиҙӘеҝғйҖүжӢ©еҗҺпјҢеҺҹй—®йўҳжҖ»жҳҜжңүжңҖдјҳи§ЈпјҢжүҖд»ҘиҙӘеҝғйҖүжӢ©жҖ»жҳҜе®үе…Ёзҡ„гҖӮ
  3. иҜҒжҳҺеҒҡеҮәиҙӘеҝғйҖүжӢ©еҗҺзҡ„еӯҗй—®йўҳе’ҢиҙӘеҝғйҖүжӢ©дёҖиө·жһ„жҲҗеҺҹй—®йўҳзҡ„жңҖдјҳи§ЈпјҢеҚіеҺҹй—®йўҳе…·жңүжңҖдјҳеӯҗз»“жһ„гҖӮ

иғҪеӨҹз”ЁиҙӘеҝғз®—жі•жұӮи§Јзҡ„й—®йўҳе…·жңүиҙӘеҝғйҖүжӢ©жҖ§иҙЁе’ҢжңҖдјҳеӯҗз»“жһ„гҖӮ

иҙӘеҝғйҖүжӢ©жҖ§иҙЁ(Greedy-choice property)

иҙӘеҝғйҖүжӢ©жҖ§иҙЁ(greedy-choice property)пјҡеҸҜд»ҘйҖҡиҝҮдҪңеҮәеұҖйғЁжңҖдјҳпјҲиҙӘеҝғпјүйҖүжӢ©жқҘжһ„йҖ е…ЁеұҖжңҖдјҳи§ЈгҖӮ

[!important]

иҙӘеҝғз®—жі•е’ҢеҠЁжҖҒи§„еҲ’зҡ„еҜ№жҜ”пјҡ

  • еҠЁжҖҒи§„еҲ’зҡ„йҖүжӢ©дҫқиө–дәҺеӯҗй—®йўҳзҡ„и§ЈпјҢжүҖд»ҘдёҖиҲ¬йҮҮз”ЁиҮӘеә•еҗ‘дёҠзҡ„ж–№жі•жұӮи§Јй—®йўҳгҖӮ
  • иҙӘеҝғз®—жі•дёҚдҫқиө–дәҺеӯҗй—®йўҳзҡ„и§ЈпјҢжүҖд»ҘдёҖиҲ¬йҮҮз”ЁиҮӘйЎ¶еҗ‘дёӢзҡ„ж–№жі•жұӮи§Јй—®йўҳгҖӮ
  • еӣ жӯӨеҝ…йЎ»иҜҒжҳҺжҜҸдёӘжӯҘйӘӨеҒҡеҮәиҙӘеҝғйҖүжӢ©иғҪз”ҹжҲҗе…ЁеұҖжңҖдјҳи§Јпјҡ
    • йҰ–е…ҲиҖғжҹҘжҹҗдёӘеӯҗй—®йўҳзҡ„жңҖдјҳи§ЈпјҢз„¶еҗҺз”ЁиҙӘеҝғйҖүжӢ©жӣҝжҚўжҹҗдёӘе…¶д»–йҖүжӢ©жқҘдҝ®ж”№жӯӨи§ЈпјҢд»ҺиҖҢеҫ—еҲ°дёҖдёӘзӣёдјјдҪҶжӣҙе°Ҹзҡ„еӯҗй—®йўҳгҖӮ

еҰӮжһңжҲ‘д»¬иҝӣиЎҢиҙӘеҝғйҖүжӢ©ж—¶дёҚеҫ—дёҚиҖғиҷ‘дј—еӨҡйҖүжӢ©пјҢжҲ‘д»¬еҝ…йЎ»жҸҗй«ҳиҝӣиЎҢиҙӘеҝғйҖүжӢ©зҡ„ж•ҲзҺҮгҖӮйҖҡиҝҮеҜ№иҫ“е…Ҙж•°жҚ®иҝӣиЎҢйў„еӨ„зҗҶжҲ–иҖ…йҖүжӢ©еҗҲйҖӮзҡ„ж•°жҚ®з»“жһ„пјҲйҖҡеёёжҳҜдјҳе…ҲйҳҹеҲ—пјүпјҢжҲ‘д»¬еҸҜд»Ҙеҝ«йҖҹеҒҡеҮәиҙӘеҝғйҖүжӢ©пјҢд»ҺиҖҢеҫ—еҲ°й«ҳж•Ҳзҡ„з®—жі•гҖӮ

жңҖдјҳеӯҗз»“жһ„(Optimal substructure)

[!note]

еҰӮжһңдёҖдёӘй—®йўҳзҡ„жңҖдјҳи§ЈеҢ…еҗ«е…¶еӯҗй—®йўҳзҡ„жңҖдјҳи§ЈпјҢеҲҷз§°жӯӨй—®йўҳе…·жңүжңҖдјҳеӯҗз»“жһ„жҖ§иҙЁгҖӮ

иҙӘеҝғз®—жі•йҖҡеёёйҮҮз”ЁдёҖдёӘйқһеёёзӣҙжҺҘзҡ„ж–№жі•иҝҗз”ЁжңҖдјҳеӯҗз»“жһ„пјҢеҒҮе®ҡйҖҡиҝҮеҜ№еҺҹй—®йўҳзҡ„иҙӘеҝғйҖүжӢ©е°ұеҸҜд»Ҙеҫ—еҲ°еӯҗй—®йўҳпјҢжҲ‘д»¬еҸӘйңҖиҰҒиҜҒжҳҺеҒҡеҮәиҙӘеҝғйҖүжӢ©еҗҺзҡ„еӯҗй—®йўҳе’ҢиҙӘеҝғйҖүжӢ©дёҖиө·жһ„жҲҗеҺҹй—®йўҳзҡ„жңҖдјҳи§ЈгҖӮ

иҜҘж–№жі•йҡҗејҸең°иҝҗз”ЁдәҶеҪ’зәіжі•иҜҒжҳҺдәҶжҜҸдёҖжӯҘиҙӘеҝғйҖүжӢ©йғҪиғҪжһ„жҲҗеҺҹй—®йўҳзҡ„жңҖдјҳи§ЈгҖӮ

иҙӘеҝғVSеҠЁжҖҒи§„еҲ’(Greedy versus dynamic programming)

иҷҪз„¶иҙӘеҝғе’ҢеҠЁжҖҒи§„еҲ’ж–№жі•йғҪеҜ№й—®йўҳзҡ„жңҖдјҳеӯҗз»“жһ„иҝӣиЎҢдәҶз ”з©¶пјҢдҪҶжҳҜеҰӮдҪ•йҖүжӢ©дёӨз§Қж–№жі•жұӮи§Јй—®йўҳеҚҙжҳҜдёӘйҡҫзӮ№пјҢжүҖд»ҘжҲ‘д»¬з”ЁдёӢйқўдёӨдёӘй—®йўҳиҝӣиЎҢдёҫдҫӢгҖӮ

0-1иғҢеҢ…й—®йўҳ(0-1 knapsack problem)пјҡе•Ҷеә—йҮҢжңү n дёӘе•Ҷе“ҒпјҢз¬¬ i дёӘе•Ҷе“Ғд»·еҖјдёә vi пјҢйҮҚйҮҸдёә wi пјҢзҺ°еңЁжңүдёҖдёӘеҸҜе®№зәіжңҖеӨ§йҮҚйҮҸдёә W зҡ„иғҢеҢ…пјҢзӣ®ж ҮдҪҝеҫ—иғҢеҢ…дёӯзҡ„е•Ҷе“ҒжҖ»д»·еҖјжңҖеӨ§пјҢжҜҸдёӘе•Ҷе“ҒеҸӘиғҪйҖүжӢ©дёҖж¬ЎпјҢдё”иҰҒдёҚйҖүжӢ©ж•ҙдёӘжҹҗе•Ҷе“ҒиҰҒдёҚдёҚйҖүжӢ©жҹҗе•Ҷе“ҒпјҢеә”иҜҘйҖүжӢ©е“Әдәӣе•Ҷе“ҒгҖӮ

еҲҶж•°иғҢеҢ…й—®йўҳ(fractional knapsack problem)пјҡй—®йўҳеңәжҷҜе’Ң0-1иғҢеҢ…й—®йўҳдёҖж ·пјҢе”ҜдёҖеҢәеҲ«жҳҜеҸҜд»ҘйҖүжӢ©жӢҝиө°е•Ҷе“Ғзҡ„дёҖйғЁеҲҶпјҢеә”иҜҘйҖүжӢ©е“Әдәӣе•Ҷе“ҒгҖӮ

еҸҜд»Ҙе°Ҷ0-1 иғҢеҢ…й—®йўҳдёӯзҡ„е•Ҷе“ҒжғіиұЎдёәйҮ‘й”ӯпјҢиҖҢеҲҶж•°иғҢеҢ…й—®йўҳдёӯзҡ„е•Ҷе“ҒжӣҙеғҸйҮ‘з ӮгҖӮ

дёӨдёӘиғҢеҢ…й—®йўҳ(knapsack problem)йғҪе…·жңүжңҖдјҳеӯҗз»“жһ„гҖӮ

еҲҶж•°иғҢеҢ…й—®йўҳеҸҜд»Ҙз”ЁиҙӘеҝғзӯ–з•ҘжұӮи§ЈпјҢз»ғд№ 15.2-1иҰҒжұӮдҪ иҜҒжҳҺеҲҶж•°иғҢеҢ…й—®йўҳзҡ„иҙӘеҝғйҖүжӢ©жҖ§иҙЁгҖӮйҰ–е…Ҳи®Ўз®—жҜҸдёӘе•Ҷе“Ғзҡ„жҜҸзЈ…д»·еҖјv./w,гҖӮйҒөеҫӘиҙӘеҝғзӯ–з•ҘпјҢе°ҸеҒ·е°ҪйҮҸеӨҡең°жӢҝиө°жҜҸзЈ…д»·еҖјжңҖй«ҳзҡ„е•Ҷе“Ғ

0-1иғҢеҢ…й—®йўҳеҸӘиғҪз”ЁеҠЁжҖҒи§„еҲ’жұӮи§ЈпјҢз»ғд№ 15.2-2иҰҒжұӮдҪ з”ЁеҠЁжҖҒи§„еҲ’жұӮи§Ј0-1иғҢеҢ…й—®йўҳгҖӮеӣ дёәеҸҜиғҪж— жі•жі•иЈ…ж»ЎиғҢеҢ…пјҢз©әй—Із©әй—ҙйҷҚдҪҺдәҶжЎҲзҡ„жңүж•ҲжҜҸзЈ…д»·еҖјеҜјиҮҙиҙӘеҝғзӯ–з•ҘеӨұиҙҘ

[!note]

з»ҷе®ҡдёӨдёӘйӣҶеҗҲ A е’Ң B пјҢжҜҸдёӘйӣҶеҗҲйғҪеҢ…еҗ« n дёӘжӯЈж•ҙж•°пјҢдҪ еҸҜд»ҘжҢүз…§д»»ж„ҸйЎәеәҸйҮҚжҺ’жҜҸдёӘйӣҶеҗҲгҖӮйҮҚжҺ’еҗҺпјҢд»Ө ai дёәйӣҶеҗҲ A дёӯз¬¬ i дёӘе…ғзҙ пјҢ bi дёәйӣҶеҗҲ B дёӯз¬¬ i дёӘе…ғзҙ пјҢиҜ·и®ҫи®ЎдёҖдёӘз®—жі•пјҢдҪҝеҫ—ai^biзҡ„е’ҢжңҖеӨ§еҢ–пјҢз»ҷеҮәиҜҘз®—жі•зҡ„иҝҗиЎҢж—¶й—ҙпјҢе№¶иҜҒжҳҺиҜҘз®—жі•зҡ„жӯЈзЎ®жҖ§гҖӮ
